create table ProfileStyle
(
    id           int auto_increment
        primary key,
    trainerEmail varchar(320) not null,
    styleID      int          not null,
    constraint ProfileStyle_TrainerStyle_id_fk
        foreign key (styleID) references TrainerStyle (id),
    constraint ProfileStyle_ibfk_1
        foreign key (trainerEmail) references Trainer (email)
);

create index profileEmail
    on ProfileStyle (trainerEmail);

create index styleID
    on ProfileStyle (styleID);
