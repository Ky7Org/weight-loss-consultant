create table Package
(
    id                  int auto_increment
        primary key,
    exercisePlan        varchar(1000) charset utf8 null,
    schedule            varchar(1000) charset utf8 not null,
    price               float                      not null,
    trainerEmail        varchar(320)               null,
    status              int                        not null,
    dietPlan            varchar(1000) charset utf8 not null,
    spendTimeToTraining int                        not null,
    name                varchar(1000)              null,
    createDate          float                      null,
    sessionLength       int                        null,
    endDate             bigint                     null,
    startDate           bigint                     null,
    constraint Package_ibfk_1
        foreign key (trainerEmail) references Trainer (email)
);

create index status
    on Package (status);

create index trainerEmail
    on Package (trainerEmail);
