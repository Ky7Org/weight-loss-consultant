create table HealthInformation
(
    id             int auto_increment
        primary key,
    weight         float                      not null,
    height         float                      not null,
    medicalHistory varchar(1000) charset utf8 not null,
    exerciseHabit  varchar(1000) charset utf8 not null,
    diet           varchar(1000) charset utf8 not null,
    customerEmail  varchar(320)               not null,
    timestamp      bigint                     not null,
    constraint HealthInformation_Customer_email_fk
        foreign key (customerEmail) references Customer (email)
);
