create table Report
(
    id                  int auto_increment
        primary key,
    contractID          int                        not null,
    exerciseDescription varchar(1000) charset utf8 not null,
    dietDescription     varchar(1000) charset utf8 not null,
    trainerFeedback     varchar(1000)              null,
    trainerApproval     int                        null,
    weight              int                        not null,
    createDate          bigint                     null,
    constraint Report_Contract_id_fk
        foreign key (contractID) references Contract (id)
);
