create table Contract
(
    id            int auto_increment
        primary key,
    campaignID    int   not null,
    packageID     int   not null,
    totalPrice    float not null,
    timeOfExpired float not null,
    timeOfCreate  float not null,
    status        int   not null,
    constraint Contract_Campaign_id_fk
        foreign key (campaignID) references Campaign (id),
    constraint Contract_Package_id_fk
        foreign key (packageID) references Package (id)
);

create index campaignID
    on Contract (campaignID);

create index packageID
    on Contract (packageID);
