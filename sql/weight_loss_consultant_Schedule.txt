create table Schedule
(
    id             int    not null
        primary key,
    packageID      int    not null,
    startTrainTime bigint not null,
    endTrainTime   bigint not null,
    contractID     int    not null,
    constraint Schedule_Contract_id_fk
        foreign key (contractID) references Contract (id)
);

create index Schedule_ibfk_1
    on Schedule (contractID);

create index packageID
    on Schedule (packageID);
