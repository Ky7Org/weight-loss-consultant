create table Campaign
(
    id                   int auto_increment
        primary key,
    customerEmail        varchar(320)               not null,
    description          varchar(1000) charset utf8 null,
    status               int                        not null,
    startDate            bigint                     not null,
    endDate              bigint                     not null,
    feedback             varchar(1000) charset utf8 null,
    targetWeight         float                      not null,
    currentWeight        float                      not null,
    spendTimeForTraining float                      not null,
    createDate           float                      null,
    constraint Campaign_Customer_email_fk
        foreign key (customerEmail) references Customer (email)
);

create index status
    on Campaign (status);
